create table if not exists public.marathon_data (
  user_id uuid primary key references auth.users(id) on delete cascade,
  data jsonb not null default '{}'::jsonb,
  updated_at timestamptz not null default now()
);
alter table public.marathon_data enable row level security;
drop policy if exists "own read" on public.marathon_data;
drop policy if exists "own insert" on public.marathon_data;
drop policy if exists "own update" on public.marathon_data;
create policy "own read" on public.marathon_data for select to authenticated using (auth.uid() = user_id);
create policy "own insert" on public.marathon_data for insert to authenticated with check (auth.uid() = user_id);
create policy "own update" on public.marathon_data for update to authenticated using (auth.uid() = user_id) with check (auth.uid() = user_id);
